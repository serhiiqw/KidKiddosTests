95       public boolean finalPrice(){
96           double pricePerBook = 36.99;
97           double previousPrice = pricePerBook*5;
98   
99           WebElement updatedPrice = findElementByXpath("//span[contains(@class, 'currency-converter-amount')]");
100          String finalUpdatedPrice = updatedPrice.getText().replace("CAD", "").replace(",", "").trim();
101  
102          double actualTotalPrice;
103          try {
104              actualTotalPrice = Double.parseDouble(finalUpdatedPrice);
105          } catch (NumberFormatException e) {
106              // Если текст не может быть преобразован в число, возвращаем false
107              e.printStackTrace();
108              return false;
109          }
110          return actualTotalPrice != previousPrice;
111      }
