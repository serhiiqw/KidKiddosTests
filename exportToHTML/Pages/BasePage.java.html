1    package Pages;
2    import org.openqa.selenium.By;
3    import org.openqa.selenium.WebDriver;
4    import org.openqa.selenium.WebElement;
5    import org.openqa.selenium.support.ui.ExpectedConditions;
6    import org.openqa.selenium.support.ui.WebDriverWait;
7    
8    import java.time.Duration;
9    
10   public class BasePage {
11   
12       protected static WebDriver webDriver;
13       protected static WebDriverWait wait;
14   //Эта функция связывает UseCase abd BasePage только созданием драйвера и ожиданием
15       public void setWebDriver(WebDriver webDriver){
16           this.webDriver = webDriver;
17           wait = new WebDriverWait(webDriver, Duration.ofSeconds(5));
18       }
19   
20       //Первую функция я создал поиск по ByXpath, так как в дальнейшем проще работать с другими методами
21       protected WebElement findElementByXpath(String xpath){ //Чтобы найти мы используем аргумент Xpath
22           WebElement element; //Здесь мы не присваиваем, что может запутать, а просто создаем элемент, который нашли
23           wait.until(ExpectedConditions.presenceOfAllElementsLocatedBy(By.xpath(xpath))); //Смотри в первую функцию и ждем пока загрузиться элемент
24           element = webDriver.findElement(By.xpath(xpath)); //Присваиваем найденный элемент переменной
25           return element; //Возвращаем его
26       }
27   
28       protected void clickByXpath(String xpath){
29           findElementByXpath(xpath).click();
30       }
31   
32       protected void sendTextToElement(String xpath, String text){
33           findElementByXpath(xpath).sendKeys(text);
34       }
35   
36       protected boolean elementExists(String xpath){
37           try{
38               webDriver.findElement(By.xpath(xpath));
39               return true;
40           }
41           catch (Exception err){
42               return false;
43           }
44       }
45   
46   
47   }
